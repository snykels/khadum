-- استعلام 2: فحص التكرار
SELECT 
  phone, 
  COUNT(*) AS cnt, 
  array_agg(id) AS user_ids,
  array_agg(name) AS names,
  array_agg(role) AS roles
FROM users 
WHERE phone IS NOT NULL 
GROUP BY phone 
HAVING COUNT(*) > 1;


-- استعلام 3: فحص الصيغ غير المتوافقة
SELECT 
  id, 
  name, 
  role,
  phone,
  length(phone) AS phone_length,
  phone ~ '^966[0-9]{9}$' AS is_normalized
FROM users 
WHERE phone IS NOT NULL 
  AND phone !~ '^966[0-9]{9}$'
ORDER BY role, id;


-- استعلام 4 (إضافي): توزيع الصيغ الموجودة
SELECT 
  CASE 
    WHEN phone ~ '^966[0-9]{9}$' THEN 'normalized (966XXXXXXXXX)'
    WHEN phone ~ '^\+966[0-9]{9}$' THEN 'with_plus (+966)'
    WHEN phone ~ '^00966[0-9]{9}$' THEN 'with_00966'
    WHEN phone ~ '^0[0-9]{9}$' THEN 'local_with_zero (05X)'
    WHEN phone ~ '^5[0-9]{8}$' THEN 'local_no_zero (5X)'
    WHEN phone IS NULL THEN 'null'
    ELSE 'other/invalid'
  END AS format_type,
  COUNT(*) AS count
FROM users
GROUP BY format_type
ORDER BY count DESC;
